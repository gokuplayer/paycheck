using System;

class Program
{
    static void Main() {
        
        float wages = 17f;
        float hoursWorked = 16;
        float federalIncomeTax = 6.97f;
        float stateIncomeTax = 2.93f;
        float localIncomeTax = 0f;
        float socialSecurity = 6.2f;
        float medicare = 1.45f;
        float retirementPlan = 3f;
        float grossIncome;
        float totalTaxPercent;
        float totalTaxes;
        float netIncome;
        
        grossIncome = wages * hoursWorked;
        totalTaxPercent = (federalIncomeTax + stateIncomeTax + localIncomeTax + socialSecurity + medicare + retirementPlan) / 100;
        totalTaxes = grossIncome * totalTaxPercent;
        netIncome = grossIncome - totalTaxes;
        
        Console.WriteLine("Predicted Gross Income: " + grossIncome.ToString());
        Console.WriteLine("Predicted Tax Percent: " + totalTaxPercent.ToString());
        Console.WriteLine("Predicted Taxes Taken Out: " + totalTaxes.ToString());
        Console.WriteLine("Predicted Paycheck: " + netIncome.ToString());
        
    }
}
